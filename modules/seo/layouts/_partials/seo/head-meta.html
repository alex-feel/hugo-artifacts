{{/* seo/head-meta.html -- renderer (RENDERER tier). Purpose: write the classic head surface tag-by-tag in a fixed order: , description, canonical, robots (+ per-bot metas), hreflang block (+ x-default), Open Graph block (+ og:type extras keyed on ogType), Twitter Card block, verification block, and feed discovery. No JSON-LD is emitted here. Expected context: $ctx = {page, seo} as assembled by seo/head.html. Return value: none. This partial renders. 404 handling: a 404 page has no canonical URL, so canonical and hreflang are suppressed; the robots resolver has already forced noindex. Multilingual output is gated on hugo.IsMultilingual. */}} {{- $page := .page -}} {{- $seo := .seo -}} {{- $is404 := $seo.is404 -}} {{/* ---- Title ---- <title> uses titleFull (adds the site title suffix); OG/JSON-LD use title, because the suffix is a SERP display device, never part of the entity's canonical name. */}} <title>{{ $seo.titleFull }} {{/* ---- Description ---- the unified description shared with OG/Twitter/JSON-LD */}} {{- with $seo.description }} {{- end }} {{/* ---- Canonical ---- absolute URL, suppressed on 404 (no canonical URL) */}} {{- if not $is404 }} {{- with $seo.canonical }} {{- end }} {{- end }} {{/* ---- Robots ---- merged all-bots directive plus the per-bot map */}} {{- with $seo.robots }} {{- end }} {{- range $bot, $directive := $seo.robotsBots }} {{- with $directive }} {{- end }} {{- end }} {{/* ---- hreflang + x-default ---- Suppressed on 404 and when the current page itself is noindex. Each hreflang value is .Language.Locale, an RFC 5646 language-TERRITORY tag (e.g. en-US, de-DE). .AllTranslations includes the current page, so one range yields the full symmetric set. A translation whose own resolved robots contains noindex is skipped so Google never sees an hreflang pointing at a noindexed URL. x-default targets the DEFAULT-language variant selected via .Language.IsDefault (never self-pointing); when no default-language translation exists it is omitted, never fabricated. */}} {{- $currentNoindex := in (lower $seo.robots) "noindex" -}} {{- if and (not $is404) (not $currentNoindex) (hugo.IsMultilingual) -}} {{- range $page.AllTranslations -}} {{- $tRobots := (partial "seo/resolve/robots.html" .).robots -}} {{- if not (in (lower $tRobots) "noindex") }} {{- end -}} {{- end -}} {{- $default := "" -}} {{- range $page.AllTranslations -}} {{- if .Language.IsDefault -}} {{- $default = . -}} {{- end -}} {{- end -}} {{- with $default }} {{- end -}} {{- end }} {{/* ---- Open Graph ---- og:site_name from organization name -> website name -> site.Title. og:image carries real width/height from resolve/image.html; og:image:alt is ALWAYS present when an image is (ogp.me requires it, and many implementations omit it). The first image is the primary card. */}} {{- $ogSiteName := $page.Param "seo.organization.name" | default ($page.Param "seo.website.name") | default site.Title -}} {{- with $seo.description }} {{- end }} {{- with $ogSiteName }} {{- end }} {{- with $seo.locale }} {{- end }} {{- if hugo.IsMultilingual }} {{- range $page.Translations }} {{- end }} {{- end }} {{- with (index $seo.images 0) }} {{- with .type }} {{- end }} {{- with .width }} {{- end }} {{- with .height }} {{- end }} {{- with .alt }} {{- end }} {{- end }} {{/* ---- OG type extras keyed on ogType ---- article:* on article; profile:* on profile; product:price:* on product; og:video:* on video.other. Each field is guarded with `with` so empties never emit. */}} {{- if eq $seo.ogType "article" -}} {{- with $seo.datePublished }} {{- end }} {{- with $seo.dateModified }} {{- end }} {{- with $page.ExpiryDate }} {{- if not .IsZero }} {{- end }} {{- end }} {{- range $seo.authors }} {{- with .url }} {{- end }} {{- end }} {{- $section := "" -}} {{- with $page.Params.categories -}} {{- $section = index . 0 -}} {{- end -}} {{- if not $section -}}{{- $section = $page.Section -}}{{- end -}} {{- with $section }} {{- end }} {{- range $page.Params.tags }} {{- end }} {{- else if eq $seo.ogType "profile" -}} {{- $profile := $page.Param "seo.profile" | default (dict) -}} {{- with (index $profile "given_name") }} {{- end }} {{- with (index $profile "family_name") }} {{- end }} {{- with (index $profile "username") }} {{- end }} {{- with (index $profile "gender") }} {{- end }} {{- else if eq $seo.ogType "product" -}} {{- $offer := $page.Param "seo.offer" | default (dict) -}} {{- with (index $offer "price") }} {{- end }} {{- with (index $offer "price_currency") }} {{- end }} {{- else if eq $seo.ogType "video.other" -}} {{- $video := $page.Param "seo.video" | default (dict) -}} {{- with (index $video "content_url") }} {{- end }} {{- with (index $video "duration") }} {{- end }} {{- end }} {{/* ---- Twitter / X Card ---- Card baseline is `summary`; it upgrades to `summary_large_image` when a primary image resolves. twitter:image is emitted ONLY when a value exists (an empty value would break OG fallback); twitter:image:alt (which has no OG fallback) is emitted whenever an image is. twitter:creator = first author's twitter -> twitter_creator -> twitter_site. */}} {{- $primaryImage := index $seo.images 0 -}} {{- $twSite := $page.Param "seo.twitter_site" | default (index (site.Params) "twitter") -}} {{- $twCreator := "" -}} {{- with (index $seo.authors 0) -}} {{- with .twitter -}}{{- $twCreator = . -}}{{- end -}} {{- end -}} {{- with $twCreator -}}{{- else -}} {{- $twCreator = $page.Param "seo.twitter_creator" | default $twSite -}} {{- end -}} {{- $card := "summary" -}} {{- with $primaryImage -}}{{- $card = "summary_large_image" -}}{{- end -}} {{- with $twSite }} {{- end }} {{- with $twCreator }} {{- end }} {{- with $seo.description }} {{- end }} {{- with $primaryImage }} {{- with .alt }} {{- end }} {{- end }} {{/* ---- Site verification ---- Each token emits its provider's verification tag. Backward-compatible aliases from params.site_verification.* are read only when the seo.verification.* equivalent is absent. */}} {{- $ver := $page.Param "seo.verification" | default (dict) -}} {{- $legacyVer := index (site.Params) "site_verification" | default (dict) -}} {{- with (index $ver "google") | default (index $legacyVer "google") }} {{- end }} {{- with (index $ver "bing") | default (index $legacyVer "bing") }} {{- end }} {{- with (index $ver "baidu") | default (index $legacyVer "baidu") }} {{- end }} {{- with (index $ver "facebook") | default (index $legacyVer "facebook") }} {{- end }} {{- with (index $ver "pinterest") }} {{- end }} {{- with (index $ver "mastodon") | default (index $legacyVer "mastodon") }} {{- end }} {{/* ---- Feed discovery ---- Emit for the page's RSS output when enabled and present. Filter .AlternativeOutputFormats to the RSS format so JSON and other alternates are never mislabeled as feeds. When the page itself has no RSS format (a leaf page), fall back to the page's section RSS. */}} {{- if ne ($page.Param "seo.feed.enable") false -}} {{- $feedTitle := $ogSiteName -}} {{- $rss := $page.AlternativeOutputFormats.Get "rss" -}} {{- with $rss }} {{- else -}} {{- with $page.CurrentSection -}} {{- with .OutputFormats.Get "rss" }} {{- end -}} {{- end -}} {{- end -}} {{- end }}
