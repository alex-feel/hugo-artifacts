{{/* seo/jsonld/videoobject.html -- VideoObject node builder (self-gates to name + thumbnailUrl + uploadDate). Purpose: build the VideoObject content node. Expected context (dict): - .page (Page) the current page. - .seo (dict) the resolved $seo context. Self-gating: `page` kind AND name AND thumbnailUrl AND uploadDate all resolve. Any of the three missing warns and suppresses the node; lists/home suppress silently by kind. Mutual-exclusion and uniqueness rules: - hasPart[] Clip and potentialAction SeekToAction are mutually exclusive; clips win and a warn fires when both are set. - seo.video.clips must be a LIST of clip tables: a non-list value is ignored with a warn (and seek_url_template, when set, then applies), and a list entry that is not a table is skipped with the same per-clip warn as a missing property -- author input never crashes the build. - Every Clip requires name, start_offset, and url (all three are Google-required for Key Moments): a clip missing any of them, or whose start_offset does not parse, is skipped WHOLE with a warn (lib/num.html warns the unparseable case) -- fabricating a default for a missing value could mislabel a key moment. end_offset is the only optional clip property; an unparseable value drops just that property. - The WHOLE hasPart is dropped (with a warn) when two ELIGIBLE clips share a start_offset (Key Moments require unique start offsets; a skipped clip cannot collide); when every clip is skipped, hasPart is not emitted. - regionsAllowed XOR ineligibleRegion; both set warns and drops ineligible_region. Return value: the VideoObject node dict WITHOUT @context, or an EMPTY dict (dict) when ineligible. One terminal return. */}} {{- $ctx := . -}} {{- $page := $ctx.page -}} {{- $seo := $ctx.seo -}} {{- $result := dict -}} {{- $video := $page.Param "seo.video" | default dict -}} {{- if eq $page.Kind "page" -}} {{- $name := $seo.title -}} {{/* Deprecated video_id alias: a bare YouTube id seeds seo.video.embed_url and seo.video.thumbnail_url; the explicit seo.video keys always win, so the alias fills only the gaps. */}} {{- $videoId := $page.Params.video_id -}} {{/* thumbnailUrl[]: seo.video.thumbnail_url -> video_id-derived YouTube thumbnail (alias) -> images[] URLs. */}} {{- $thumbs := slice -}} {{- with $video.thumbnail_url -}} {{- if reflect.IsSlice . -}} {{- range . -}} {{- $thumbs = $thumbs | append . -}} {{- end -}} {{- else -}} {{- $thumbs = $thumbs | append . -}} {{- end -}} {{- end -}} {{- if not $thumbs -}} {{- with $videoId -}} {{- $thumbs = $thumbs | append (printf "https://i.ytimg.com/vi/%s/hqdefault.jpg" .) -}} {{- end -}} {{- end -}} {{- if not $thumbs -}} {{- if and $seo.images (gt (len $seo.images) 0) -}} {{- range $seo.images -}} {{- with .url -}} {{- $thumbs = $thumbs | append . -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{/* uploadDate: seo.video.upload_date -> the resolved $seo.datePublished (which itself falls back to .Date). */}} {{- $uploadDate := "" -}} {{- with $video.upload_date -}} {{- $uploadDate = partial "seo/lib/time.html" (dict "value" . "field" "seo.video.upload_date" "page" $page) -}} {{- end -}} {{- if not $uploadDate -}} {{- $uploadDate = $seo.datePublished -}} {{- end -}} {{/* Self-gate: all three required properties. */}} {{- $missing := slice -}} {{- if not $name -}}{{- $missing = $missing | append "name" -}}{{- end -}} {{- if not $thumbs -}}{{- $missing = $missing | append "thumbnailUrl" -}}{{- end -}} {{- if not $uploadDate -}}{{- $missing = $missing | append "uploadDate" -}}{{- end -}} {{- if gt (len $missing) 0 -}} {{- partial "seo/lib/warn.html" (dict "key" (printf "seo:warn:video-required:%s" $page.RelPermalink) "message" (printf "[seo] VideoObject on %q is missing %s; Google requires name, thumbnailUrl, and uploadDate. Node not emitted." $page.RelPermalink (delimit $missing ", ")) ) -}} {{- else -}} {{- $node := dict "@type" "VideoObject" "@id" $seo.ids.content "name" $name "thumbnailUrl" $thumbs "uploadDate" $uploadDate -}} {{- with $seo.description -}} {{- $node = merge $node (dict "description" .) -}} {{- end -}} {{- with $video.content_url -}} {{- $node = merge $node (dict "contentUrl" .) -}} {{- end -}} {{/* embedUrl: seo.video.embed_url -> video_id-derived YouTube player URL (alias). */}} {{- $embed := $video.embed_url -}} {{- if not $embed -}} {{- with $videoId -}} {{- $embed = printf "https://www.youtube.com/embed/%s" . -}} {{- end -}} {{- end -}} {{- with $embed -}} {{- $node = merge $node (dict "embedUrl" .) -}} {{- end -}} {{- with $video.duration -}} {{- $node = merge $node (dict "duration" .) -}} {{- end -}} {{- with $video.expires -}} {{- $expires := partial "seo/lib/time.html" (dict "value" . "field" "seo.video.expires" "page" $page) -}} {{- with $expires -}} {{- $node = merge $node (dict "expires" .) -}} {{- end -}} {{- end -}} {{/* hasPart[] Clip (Key Moments) XOR potentialAction SeekToAction. Clips win the deterministic tiebreak when both are set, and the tiebreak warns like the other mutually-exclusive pairs so the ignored seek_url_template does not vanish silently. A clips value that is not a list (a scalar, a table) is ignored with a warn instead of crashing the range below -- and then does not suppress a valid seek_url_template. */}} {{- if and $video.clips (not (reflect.IsSlice $video.clips)) -}} {{- partial "seo/lib/warn.html" (dict "key" (printf "seo:warn:video-clips-shape:%s" $page.RelPermalink) "message" (printf "[seo] VideoObject on %q sets seo.video.clips to a value that is not a list of clip tables; clips ignored." $page.RelPermalink) ) -}} {{- end -}} {{- $hasClips := and (reflect.IsSlice $video.clips) (gt (len $video.clips) 0) -}} {{- $hasSeek := ne ($video.seek_url_template | default "") "" -}} {{- if and $hasClips $hasSeek -}} {{- partial "seo/lib/warn.html" (dict "key" (printf "seo:warn:video-clips-seek:%s" $page.RelPermalink) "message" (printf "[seo] VideoObject on %q sets both clips and seek_url_template; they are mutually exclusive. Ignoring seek_url_template." $page.RelPermalink) ) -}} {{- $hasSeek = false -}} {{- end -}} {{- if $hasClips -}} {{/* Build the eligible clip list in ONE validation pass. Google requires name, startOffset, and url on every Clip: an entry that is not a table, or a clip missing any required property, is skipped WHOLE with a warn naming what is wrong, and a clip whose start_offset does not parse is skipped via lib/num.html, which warns once naming the clip as what is omitted; an unparseable end_offset drops just that property. Nothing is fabricated for a missing value: a defaulted 0 start offset would mislabel a key moment the author never declared. Presence is isset-based, so an explicit empty value is validated (and dropped) and a native (unquoted) 0 reaches the parser. */}} {{- $clips := slice -}} {{- $clipPos := 0 -}} {{- range $video.clips -}} {{- $clipPos = add $clipPos 1 -}} {{- if not (reflect.IsMap .) -}} {{- partial "seo/lib/warn.html" (dict "key" (printf "seo:warn:video-clip-required:%s:%d" $page.RelPermalink $clipPos) "message" (printf "[seo] VideoObject on %q: clip %d is not a table of clip properties; Google requires name, startOffset, and url on every Clip. Clip skipped." $page.RelPermalink $clipPos) ) -}} {{- else -}} {{- $clipMissing := slice -}} {{- if not .name -}}{{- $clipMissing = $clipMissing | append "name" -}}{{- end -}} {{- if not (isset . "start_offset") -}}{{- $clipMissing = $clipMissing | append "start_offset" -}}{{- end -}} {{- if not .url -}}{{- $clipMissing = $clipMissing | append "url" -}}{{- end -}} {{- if gt (len $clipMissing) 0 -}} {{- partial "seo/lib/warn.html" (dict "key" (printf "seo:warn:video-clip-required:%s:%d" $page.RelPermalink $clipPos) "message" (printf "[seo] VideoObject on %q: clip %d is missing %s; Google requires name, startOffset, and url on every Clip. Clip skipped." $page.RelPermalink $clipPos (delimit $clipMissing ", ")) ) -}} {{- else -}} {{- $so := partial "seo/lib/num.html" (dict "value" .start_offset "mode" "int" "field" "seo.video.clips.start_offset" "page" $page "what" "clip") -}} {{- if ne (printf "%v" $so) "" -}} {{- $clip := dict "@type" "Clip" "name" .name "startOffset" $so "url" .url -}} {{- if isset . "end_offset" -}} {{- $eo := partial "seo/lib/num.html" (dict "value" .end_offset "mode" "int" "field" "seo.video.clips.end_offset" "page" $page) -}} {{- if ne (printf "%v" $eo) "" -}} {{- $clip = merge $clip (dict "endOffset" $eo) -}} {{- end -}} {{- end -}} {{- $clips = $clips | append $clip -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{/* Key Moments require unique start offsets: a duplicate among the ELIGIBLE clips drops the whole hasPart (choosing a survivor would be arbitrary). Skipped clips cannot collide, and eligible offsets are never negative (lib/num.html rejects negatives by default), so the -1 sentinel is unambiguous. */}} {{- $seen := dict -}} {{- $dupOffset := -1 -}} {{- range $clips -}} {{- $k := printf "%d" .startOffset -}} {{- if isset $seen $k -}} {{- if lt $dupOffset 0 -}} {{- $dupOffset = .startOffset -}} {{- end -}} {{- else -}} {{- $seen = merge $seen (dict $k true) -}} {{- end -}} {{- end -}} {{- if ge $dupOffset 0 -}} {{- partial "seo/lib/warn.html" (dict "key" (printf "seo:warn:video-clip-offset:%s" $page.RelPermalink) "message" (printf "[seo] VideoObject on %q has two clips sharing startOffset %d; Key Moments require unique start offsets. hasPart omitted." $page.RelPermalink $dupOffset) ) -}} {{- else -}} {{- with $clips -}} {{- $node = merge $node (dict "hasPart" .) -}} {{- end -}} {{- end -}} {{- else if $hasSeek -}} {{- $seekTmpl := $video.seek_url_template -}} {{- $node = merge $node (dict "potentialAction" (dict "@type" "SeekToAction" "target" (dict "@type" "EntryPoint" "urlTemplate" $seekTmpl ) "startOffset-input" "required name=seek_to_second_number" )) -}} {{- end -}} {{/* interactionStatistic (WatchAction) from watch_count, presence-gated with isset (not truthiness) so a native (unquoted) 0 reaches lib/num.html, then parsed once and reused for both the positivity guard and the emission; an unparseable count drops the property (lib/num.html warns once). The post-parse `gt $wc 0` guard is a deliberate semantic choice: a zero count is valid but not worth emitting. */}} {{- if isset $video "watch_count" -}} {{- $wc := partial "seo/lib/num.html" (dict "value" $video.watch_count "mode" "int" "field" "seo.video.watch_count" "page" $page) -}} {{- if ne (printf "%v" $wc) "" -}} {{- if gt $wc 0 -}} {{- $node = merge $node (dict "interactionStatistic" (dict "@type" "InteractionCounter" "interactionType" (dict "@type" "WatchAction") "userInteractionCount" $wc )) -}} {{- end -}} {{- end -}} {{- end -}} {{/* regionsAllowed XOR ineligibleRegion. Both set warns and drops ineligible_region (regionsAllowed wins). */}} {{- $hasRegions := and $video.regions_allowed (gt (len $video.regions_allowed) 0) -}} {{- $hasIneligible := and $video.ineligible_region (gt (len $video.ineligible_region) 0) -}} {{- if and $hasRegions $hasIneligible -}} {{- partial "seo/lib/warn.html" (dict "key" (printf "seo:warn:video-regions:%s" $page.RelPermalink) "message" (printf "[seo] VideoObject on %q sets both regionsAllowed and ineligibleRegion; keeping regionsAllowed." $page.RelPermalink) ) -}} {{- end -}} {{- if $hasRegions -}} {{- $node = merge $node (dict "regionsAllowed" $video.regions_allowed) -}} {{- else if $hasIneligible -}} {{- $node = merge $node (dict "ineligibleRegion" $video.ineligible_region) -}} {{- end -}} {{/* publication (BroadcastEvent): isLiveBroadcast + startDate + endDate. */}} {{- with $video.broadcast -}} {{- $b := $video.broadcast -}} {{- $bcast := dict "@type" "BroadcastEvent" -}} {{- $hasBcast := false -}} {{- if isset $b "is_live_broadcast" -}} {{- $bcast = merge $bcast (dict "isLiveBroadcast" $b.is_live_broadcast) -}} {{- $hasBcast = true -}} {{- end -}} {{- with $b.start_date -}} {{- $sd := partial "seo/lib/time.html" (dict "value" . "field" "seo.video.broadcast.start_date" "page" $page) -}} {{- with $sd -}} {{- $bcast = merge $bcast (dict "startDate" .) -}} {{- $hasBcast = true -}} {{- end -}} {{- end -}} {{- with $b.end_date -}} {{- $ed := partial "seo/lib/time.html" (dict "value" . "field" "seo.video.broadcast.end_date" "page" $page) -}} {{- with $ed -}} {{- $bcast = merge $bcast (dict "endDate" .) -}} {{- $hasBcast = true -}} {{- end -}} {{- end -}} {{- if $hasBcast -}} {{- $node = merge $node (dict "publication" $bcast) -}} {{- end -}} {{- end -}} {{- $result = $node -}} {{- end -}} {{- end -}} {{- return $result -}}
