{{/* Renders a privacy-first YouTube facade: a same-origin poster + a real play button that injects a youtube-nocookie player iframe ONLY on click. Nothing contacts Google before the click -- the poster is fetched at build time and republished under the site's own origin, and no preconnect/iframe-src exists at render time. Provide EITHER `id` (raw 11-char video id) OR `url` (any common YouTube URL shape). A playlist embed is also supported via `list` (or a `list=` in `url`). Parameters: - id (string) Raw 11-char video id. Wins the video-id slot over `url` when both are set; a list= / t= / start= carried in `url` is still honored, and a DIFFERENT video id in `url` is dropped with a warning. An invalid raw id fails the build even when `url` supplies a playlist. A pasted URL in `id` replaces `url` entirely, with a warning when a url was supplied alongside. - url (string) Full YouTube URL: youtu.be/, watch?v=, embed/, v/, shorts/, youtube-nocookie.com/embed/, m.youtube.com, with optional &list= and an optional ?t= / ?start= offset (the legacy #t= fragment form also works). The 11-char id is extracted and validated. - title (string) Accessible button label, iframe title, and -- when `show-title` is true -- a visible title element. - start (int) Start offset in seconds (emits start=N). Falls back to a ?t= / ?start= in `url` when this parameter is unset or invalid (a non-numeric value is warned and treated as unset). - end (int) End offset in seconds (emits end=N). - list (string) Playlist id. With no id/video, embeds the playlist (/embed/videoseries?list=...); with a video, appends list=. - poster (string) Tier-1 local poster override (page-resource name or assets path). Processed through the image pipeline. - params (string) Arbitrary extra player query string appended to the embed URL, e.g. "cc_load_policy=1&hl=fr". - loading (string) Poster loading attribute. Default "lazy". - sizes (string) Poster responsive `sizes` attribute. Default "100vw"; set it to match your layout slot to avoid over-fetching. - class (string) Extra class(es) appended to the root element. - id-anchor (string) Optional root element id (alias: `anchor`). - show-title(bool) Render a visible youtube-embed__title element. Default false. Usage: youtube-embed id="dQw4w9WgXcQ" youtube-embed url="https://youtu.be/dQw4w9WgXcQ?t=42" title="Never Gonna Give You Up" youtube-embed id="dQw4w9WgXcQ" start="42" end="90" params="cc_load_policy=1&hl=fr" youtube-embed list="PLFsQleAWXsj_4yDeebiIADdH5FMayBiJo" youtube-embed id="dQw4w9WgXcQ" poster="custom-poster.jpg" class="featured" */}} {{- $idParam := .Get "id" | default "" -}} {{- $url := .Get "url" | default "" -}} {{- $title := .Get "title" | default "" -}} {{- $startRaw := .Get "start" | default "" -}} {{- $endRaw := .Get "end" | default "" -}} {{- $listParam := .Get "list" | default "" -}} {{- $poster := .Get "poster" | default "" -}} {{- $params := .Get "params" | default "" -}} {{- $loading := .Get "loading" | default "lazy" -}} {{- $sizes := .Get "sizes" | default "100vw" -}} {{- $class := .Get "class" | default "" -}} {{- $anchor := or (.Get "id-anchor") (.Get "anchor") | default "" -}} {{/* `.Get` returns named params as strings, so "false"/"0"/"no" would all be truthy as bare non-empty strings. Coerce explicitly against accepted truthy tokens so show-title="false" actually suppresses the visible title. */}} {{- $showTitle := in (slice "true" "1" "yes" "on") (lower (.Get "show-title" | default "")) -}} {{/* Parse and validate the locator. A raw id wins the video-id slot; a url supplied alongside still contributes its list= and t=/start= values. The explicit `list` parameter wins over a url-carried list. */}} {{- $parsed := partial "youtube-embed/parse-id.html" (dict "id" $idParam "url" $url) -}} {{- $id := $parsed.id -}} {{- $list := or $listParam $parsed.list -}} {{/* A URL pasted into id= is demoted by parse-id to the url slot and REPLACES a url= supplied alongside it; surface that replacement so the discarded url (including its t=/list=) does not vanish silently. Only an explicit http:// or https:// scheme marks a pasted URL. */}} {{- $idParamIsUrl := or (hasPrefix (lower $idParam) "http://") (hasPrefix (lower $idParam) "https://") -}} {{- if and $idParamIsUrl $url -}} {{- warnf "The %q shortcode received a URL in id=%q together with url=%q; the id URL wins and url is ignored entirely, including its t=/list=. See %s" .Name $idParam $url .Position -}} {{- end -}} {{/* A supplied raw id (not a pasted URL, which parse-id demotes to the url branch) must itself validate -- a playlist id mined from url= cannot rescue a bad raw id, or a typo would silently render a playlist-only embed instead of failing loudly. */}} {{- $rawIdRejected := and $idParam (not $idParamIsUrl) (ne $parsed.id $idParam) -}} {{- if $rawIdRejected -}} {{- errorf "The %q shortcode received an invalid id=%q. YouTube ids are exactly 11 characters of [A-Za-z0-9_-]. See %s" .Name $idParam .Position -}} {{- end -}} {{/* Surface a conflicting double locator: when id= wins over a url= carrying a DIFFERENT video id, the drop is warned so a copy-paste mistake does not go unnoticed. The url's list= and t=/start= are honored either way. */}} {{- if and $idParam (not $rawIdRejected) $parsed.urlId (ne $parsed.urlId $parsed.id) -}} {{- warnf "The %q shortcode received id=%q and a url carrying a different video id %q; id wins and the url video id is ignored (the url's t=/list= are still honored). See %s" .Name $idParam $parsed.urlId .Position -}} {{- end -}} {{/* The embed is valid when parse-id resolved a usable id/list from id=/url=, OR a standalone list= parameter was supplied (a playlist-only embed). */}} {{- $valid := or $parsed.valid (ne $listParam "") -}} {{- if not $valid -}} {{- if and (not $idParam) (not $url) (not $listParam) -}} {{- errorf "The %q shortcode requires an 'id' (raw 11-char video id), a 'url' (any YouTube URL), or a 'list' (playlist id). See %s" .Name .Position -}} {{- else -}} {{- errorf "The %q shortcode could not extract a valid 11-character YouTube video id from id=%q url=%q. YouTube ids are exactly 11 characters of [A-Za-z0-9_-]. See %s" .Name $idParam $url .Position -}} {{- end -}} {{- end -}} {{/* Normalize start/end to non-negative integer seconds HERE, so the facade receives clean ints and never risks an `int "abc"` build break. A non-numeric or negative value is ignored (treated as unset) with a single warning -- an optional timing hint must not fail the build. Leading zeros are stripped so the base-0 cast cannot read a zero-padded value as octal (int "011" -> 9) or die on an invalid octal digit (int "08" is fatal); an all-zero value trims to empty and keeps the 0 default, and the try absorbs an overflowing digit run. $startSet tracks whether a VALID explicit start was supplied, so an invalid value stays "treated as unset" and the url-carried offset below still applies. */}} {{- $start := 0 -}} {{- $startSet := false -}} {{- with $startRaw -}} {{- if findRE `^[0-9]+$` . 1 -}} {{- $startSet = true -}} {{- $s := strings.TrimLeft "0" . -}} {{- if $s -}} {{- with try (int $s) -}} {{- with .Value -}} {{- $start = . -}} {{- end -}} {{- end -}} {{- end -}} {{- else -}} {{- warnf "The %q shortcode ignored a non-numeric start=%q (expected whole seconds). See %s" $.Name . $.Position -}} {{- end -}} {{- end -}} {{- $end := 0 -}} {{- with $endRaw -}} {{- if findRE `^[0-9]+$` . 1 -}} {{- $e := strings.TrimLeft "0" . -}} {{- if $e -}} {{- with try (int $e) -}} {{- with .Value -}} {{- $end = . -}} {{- end -}} {{- end -}} {{- end -}} {{- else -}} {{- warnf "The %q shortcode ignored a non-numeric end=%q (expected whole seconds). See %s" $.Name . $.Position -}} {{- end -}} {{- end -}} {{/* A start offset carried in the URL (?t= / ?start=, e.g. youtu.be/ID?t=42) is honored only when no VALID explicit `start` parameter was supplied; the valid explicit parameter always wins, and an invalid one counts as unset. */}} {{- if and (not $startSet) (gt $parsed.start 0) -}} {{- $start = $parsed.start -}} {{- end -}} {{/* Assemble the shared context. `page` and `position` are carried for the poster resolution (page-resource lookup, warn-dedup) and diagnostics. */}} {{- $ctx := dict "id" $id "list" $list "title" $title "start" $start "end" $end "params" $params "loading" $loading "sizes" $sizes "showTitle" $showTitle "anchor" $anchor "poster" $poster "page" .Page "position" .Position -}} {{/* Resolve the poster (T1 local -> T2 maxres -> T3 hq -> T4 default -> none). */}} {{- $posterData := partial "youtube-embed/poster.html" $ctx -}} {{/* Merge poster result + extra class, then render the facade. */}} {{- $data := merge $ctx (dict "poster" $posterData "extraClass" $class) -}} {{- partial "youtube-embed/facade.html" $data -}}
