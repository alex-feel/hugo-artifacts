{{/* Converts an ISO 8601 timestamp to a human-readable relative time string. Fed from the Space's lastModified field on the Hugging Face Hub. Identical formatting contract to the github-repo sibling module. Context: an ISO 8601 date string (e.g., "2026-04-21T10:30:00Z") Returns a string like "3 days ago", "2 months ago", "just now", or "" when the timestamp is unparseable -- the input is remote-derived, so a malformed value degrades to an empty string instead of failing the build. Every phrase resolves through the shipped hf_space_* i18n keys (plural forms selected by CLDR rules from the integer count) with English defaults, so a site language without the keys still renders. */}} {{- $result := "" -}} {{- $t := try (time.AsTime .) -}} {{- $hours := 0.0 -}} {{- $parsedOk := false -}} {{- if not $t.Err -}} {{- with $t.Value -}} {{- $hours = (now.Sub .).Hours -}} {{- $parsedOk = true -}} {{- end -}} {{- end -}} {{- if not $parsedOk -}} {{- else if lt $hours 1.0 -}} {{- $result = T "hf_space_just_now" | default "just now" -}} {{- else if lt $hours 24.0 -}} {{- $n := int (math.Floor $hours) -}} {{- $result = T "hf_space_hours_ago" $n | default (cond (eq $n 1) "1 hour ago" (printf "%d hours ago" $n)) -}} {{- else if lt $hours 48.0 -}} {{- $result = T "hf_space_yesterday" | default "yesterday" -}} {{- else if lt $hours 720.0 -}} {{- $n := int (math.Floor (math.Div $hours 24.0)) -}} {{- $result = T "hf_space_days_ago" $n | default (cond (eq $n 1) "1 day ago" (printf "%d days ago" $n)) -}} {{- else if lt $hours 8760.0 -}} {{- $n := int (math.Floor (math.Div $hours 720.0)) -}} {{- $result = T "hf_space_months_ago" $n | default (cond (eq $n 1) "1 month ago" (printf "%d months ago" $n)) -}} {{- else -}} {{- $n := int (math.Floor (math.Div $hours 8760.0)) -}} {{- $result = T "hf_space_years_ago" $n | default (cond (eq $n 1) "1 year ago" (printf "%d years ago" $n)) -}} {{- end -}} {{- return $result -}}
