{{/* Formats a number in compact notation (e.g., 1500 -> "1.5k", 1200000 -> "1.2M"). Used for the Space like count, which the Hugging Face Hub reports as a plain integer. Identical formatting contract to the github-repo sibling module so the two read as a family. Context: a numeric value (int or float) Returns a formatted string. */}} {{- $n := . -}} {{- $result := printf "%d" (int $n) -}} {{- if ge (float $n) 1000000.0 -}} {{- $d := math.Div (float $n) 1000000.0 -}} {{- $whole := math.Floor $d -}} {{- $frac := math.Round (math.Mul (math.Sub $d $whole) 10) -}} {{- if eq $frac 10.0 -}} {{- $whole = math.Add $whole 1 -}} {{- $frac = 0.0 -}} {{- end -}} {{- if eq $frac 0.0 -}} {{- $result = printf "%.0fM" $whole -}} {{- else -}} {{- $result = printf "%.0f.%.0fM" $whole $frac -}} {{- end -}} {{- else if ge (float $n) 1000.0 -}} {{- $d := math.Div (float $n) 1000.0 -}} {{- $whole := math.Floor $d -}} {{- $frac := math.Round (math.Mul (math.Sub $d $whole) 10) -}} {{- if eq $frac 10.0 -}} {{- $whole = math.Add $whole 1 -}} {{- $frac = 0.0 -}} {{- end -}} {{/* A carry that reaches 1000 crossed the tier boundary (values 999950-999999): promote to the M tier instead of rendering "1000k". The M branch above needs no such promotion -- M is the top tier. */}} {{- if ge $whole 1000.0 -}} {{- $result = "1M" -}} {{- else if eq $frac 0.0 -}} {{- $result = printf "%.0fk" $whole -}} {{- else -}} {{- $result = printf "%.0f.%.0fk" $whole $frac -}} {{- end -}} {{- end -}} {{- return $result -}}
